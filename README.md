usage exampleclass() {
  local varname="$1"
  local value="$2"
  local command="$3"

  # Устанавливаем переменную
  eval "$varname=$value"

  # Выполняем команду с использованием переменной
  eval "result=$(( $command ))"

  echo "Variable $varname = $value"
  echo "Evaluated result: $result"
}
